用户协同过滤的简单推荐算法

  • 推荐系统
  • 用户协同过滤

今天做了一个简单地推荐系统，是利用基于用户的协同过滤进行推荐的，利用的是一个电影评分的数据集，地址在这里，大致的思路为，通过用户对电影的评分，先计算用户之间的评分相似度，为了得到待推荐用户对电影的评分列表，我们利用待推荐用户与其他用户的相似度，去计算待推荐用户对电影可能的评分，然后从中选出Top-K个进行推荐。相似度度量方式可以用上一篇博文中提到的任何一个，这里我利用的是皮尔逊相似度量，对于评分的，皮尔逊度量会更为准确一些。

from math import sqrt
def sim_pearson(prefs,p1,p2):
    #相似度度量函数
    si = {}
    for item in prefs[p1]:
        if item in prefs[p2]:
            si[item] = 1
    n = len(si)

    if n == 0:
        return 0

    sum1 = sum([prefs[p1][it] for it in si])
    sum2 = sum([prefs[p2][it] for it in si])

    sum1Sq = sum([pow(prefs[p1][it], 2) for it in si])
    sum2Sq = sum([pow(prefs[p2][it], 2) for it in si])

    pSum = sum([prefs[p1][it] * prefs[p2][it] for it in si])

    num = pSum - (sum1 * sum2 / n)
    den = sqrt((sum1Sq - pow(sum1, 2) / n)*(sum2Sq - pow(sum2, 2) / n))
    if den == 0:
        return 1

    r = num / den

    return r

def getRecommendations(prefs,person,similarity=sim_pearson):
    #推荐函数
    totals = {}
    simSums = {}
    for other in prefs:
        #循环遍历其他用户
        if other == person:
            continue
        #计算其他用户与当前用户的相似度
        sim = similarity(prefs,person,other)
        if sim <= 0:
            continue
        #遍历其他用户看过的影片
        for item in prefs[other]:
        #如果当前用户没有看过该影片
            if item not in prefs[person] or prefs[person][item]==0:
                #利用相似度计算加权评分
                totals.setdefault(item,0)
                totals[item] += prefs[other][item] * sim
                #计算相似度总和
                simSums.setdefault(item,0)
                simSums[item] += sim

        #每个合计评分除以相似度总和，得到平均值
        rankings = [(total/simSums[item],item) for item, total in totals.items()]
        #排序之后翻转列表
        rankings.sort()
        rankings.reverse()
    return rankings        

def loadMovieLens(path):
    #构造数据集
    movies = {}
    movie_file = open(path+'\\Movies.csv')
    for line in movie_file:
        (id,title) = line.split(',')[0:2]
        movies[id]=title

    prefs = {}
    user_file = open(path+'\\ratings.csv')
    for line in user_file:
        (user_id,m_id,rating,ts) = line.split(',')
        prefs.setdefault(user_id,{})
        prefs[user_id][movies[m_id]] = float(rating)

    return prefs

prefs = loadMovieLens('F:\\dataset')
for i in getRecommendations(prefs,'51')[0:10]:
    print i
