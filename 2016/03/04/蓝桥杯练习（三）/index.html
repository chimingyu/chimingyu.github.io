蓝桥杯练习（三）

  • Java
  • 蓝桥杯

今天继续蓝桥杯的练习，题目是2014年第五届蓝桥杯预赛题目JAVA软件开发本科A组

  1. 猜年龄
           小明带两个妹妹参加元宵灯会。别人问她们多大了，她们调皮地说：“我们俩的年龄之积是年龄之和的6倍”。小明又补充说：“她们可不是双胞胎，年龄差肯定也不超过8岁啊。”
           请你写出：小明的较小的妹妹的年龄。
           解：直接遍历找解，小的年龄a，大的年龄b，找到的条件为a_b = 6_(a+b)，限制条件b-a<=8，解出来妹妹是10岁。

    public class A_2014_1 {
    public static void main(String args[]){
        int a=0;
        int flag = 0;
        while(true){
            a++;
            for (int i = 0;i <= 8; i++){
                if(6*(a+i+a) == a*(a+i)){
                    System.out.println(a);
                    flag = 1;
                    break;
                }
            }
            if(flag == 1)
                break;
        }
    }
}

  2. 李白打酒
            话说大诗人李白，一生好饮。幸好他从不开车。 一天，他提着酒壶，从家里出来，酒壶有酒2斗。他边走边唱：无事街上走，提壶去打酒。 逢店加一倍，遇花喝一斗。这一路上，他一共遇到店5次，遇到花10次，已知最后一次遇到的是花，他正好把酒喝光了。请你计算李白遇到店和花的次序，可以把遇店记为a，遇花记为b。则：babaabbabbabbbb 就是合理的次序。像这样的答案一共有多少呢？请你计算出所有可能方案的个数（包含题目给出的）。
           解：这道题和之前的扑克牌非常像，把代码改一改就能用了，因为最后一次要求是b，也就是说在第十四步之后，酒壶里应该只有1斗酒，那么就可以知道，第十四步肯定是遇花，这样才能保证只剩一斗酒，那么最后两个肯定都是b，这样判断前13位就好。解出来是14种。

    public class A_2014_2 {
    public static int count = 0; 
    public static void main(String [] args){
        char [] num = new char[13];
        f(num,0);
        System.out.println(count);
    }

    public static void f(char []num,int i){
        if(i >= num.length){
            if(judge(num)){
                show(num);
                count++;
            }
        }else{
            for(int j=1;j<=2;j++){
                num[i] = (char)(j + 'a' - 1);
                f(num,i+1);
            }
        }
     }

    public static void show(char [] num){
        for(int i = 0;i<num.length;i++){
            System.out.print(num[i]);
        }
        System.out.println();
    }

    public static boolean judge(char num[]){
        int countnum = 2;
        int numa=0,numb=0;
        for (char c:num){
            if(c == 'a'){
                numa++;
                countnum *= 2;
            }else{
                numb++;
                countnum--;
            }
        }

        return countnum == 2 && numa == 5 && numb == 8;
    }
}

  3. 神奇算式
    由4个不同的数字，组成的一个乘法算式，它们的乘积仍然由这4个数字组成。
    比如： 210 x 6 = 1260 8 x 473 = 3784 27 x 81 = 2187 都符合要求。如果满足乘法交换律的算式算作同一种情况，那么，包含上边已列出的3种情况，一共有多少种满足要求的算式
           解：直接遍历找解，解出来答案是12种（别忘记除2）。

    public class A_2014_3 {
    public static int count = 0; 
    public static void main(String [] args){
        for(int i = 2; i <= 999; i++){
            for(int j = 2; j <= 999; j++){
                if(judge(i,j)){
                    /*System.out.print(i);
                    System.out.print(" ");
                    System.out.print(j);
                    System.out.print(" ");
                    System.out.println(i*j);*/
                    count++;
                }
            }
        }
        System.out.println(count/2);
    }

    public static boolean judge(int num1, int num2){
        int num = num1 * num2;
        String numStr1 = String.format("%d", num);            
        String numStr2 = String.format("%d", num2) + String.format("%d", num1);
        if(judgediff(num) && judgediff(num1) && judgediff(num2) && numStr1.length() == 4 && numStr2.length() == 4){

            for (int i = 0; i < 4; i++){
                if(numStr2.indexOf(numStr1.charAt(i)) == -1){
                    return false;
                }
            }
            return true;
        }
        return false;

    }

    public static boolean judgediff(int num){
        String numStr = String.format("%d", num);
        String temp = new String("");
        for(int i = 0; i < numStr.length();i++){          
            if(temp.indexOf(numStr.charAt(i)) == -1){
                temp = temp + numStr.charAt(i);
            }else
                return false;
        }

        return true;
    }
}

  5. 锦标赛
           如果要在n个数据中挑选出第一大和第二大的数据（要求输出数据所在位置和值），使用什么方法比较的次数最少？我们可以从体育锦标赛中受到启发。如图【1.png】所示，8个选手的锦标赛，先两两捉对比拼，淘汰一半。优胜者再两两比拼…直到决出第一名。第一名输出后，只要对黄色标示的位置重新比赛即可。
    下面的代码实现了这个算法(假设数据中没有相同值)。
    代码中需要用一个数组来表示图中的树（注意，这是个满二叉树, 不足需要补齐）。它不是存储数据本身，而是存储了数据的下标。第一个数据输出后，它所在的位置被标识为-1 

    class A{ 
    //a 表示待处理的数据，长度如果不是2的次幂，则不足位置补为-1  
    static void pick(int[] a)  {   
        int n = 1; 
        while(n<a.length) 
            n *= 2;       
        int[] b = new int[2*n-1];  
        for(int i=0; i<n; i++){     
            if(i<a.length)      
                b[n-1+i] = i;    
            else     
                b[n-1+i] = -1;   
        }    
         //从最后一个向前处理 
        for(int i=b.length-1; i>0; i-=2){    
            if(b[i]<0){     
                if(b[i-1]>=0)      
                    b[(i-1)/2] = b[i-1];      
                else              
                    b[(i-1)/2] = -1;    
            }    
            else{     
                if(a[b[i]]>a[b[i-1]])      
                    b[(i-1)/2] = b[i];     
                else              
                    b[(i-1)/2] = b[i-1];    
            } 
        }    
        //输出树根 
        System.out.println(b[0] + ": " + a[b[0]]);    
        //值等于根元素的位置需要重新pk   
        pk(a,b,0,b[0]);  
        //再次输出树根 
        System.out.println(b[0] + ": " + a[b[0]]); 
     } 
  // a 表示待处理数据，b 二叉树，k 当前要重新比拼的位置，v 已经决胜出的值      
  static void pk(int[] a, int[] b, int k, int v) {    
      int k1 = k*2+1;   
      int k2 = k1 + 1;    
      if(k1>=b.length || k2>=b.length){    
          b[k] = -1;    
          return; 
      }    
      if(b[k1]==v)     
          pk(a,b,k1,v);   
      else 
          pk(a,b,k2,v);       
      //重新比较   
      if(b[k1]<0){    
          if(b[k2]>=0)     
              b[k] = b[k2];     
          else     
            b[k] = -1;    
        return;   
       }    
      if(b[k2]<0){    
          if(b[k1]>=0)     
              b[k] = b[k1];     
          else     
              b[k] = -1;    
          return;   
       }    
      if(__________________________)  //填空    
          b[k] = b[k1];   
      else 
          b[k] = b[k2];  
    } 
}
